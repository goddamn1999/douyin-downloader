📥 下载方法（选择一种）


                        

                        
                    


                    

                        

                            
# 1. 构建镜像
docker build -t douyin-downloader .

# 2. 运行（替换下面链接）
docker run -v $(pwd)/downloads:/app/Downloaded \\
  douyin-downloader \\
  python run.py -u "抖音链接"

                        

                        
                    


                    

                        

                            
# 1. 复制配置文件
cp config.example.yml config.yml

# 2. 编辑 config.yml，在 link: 下添加你的链接
# 例如：
# link:
#   - https://www.douyin.com/video/xxxxx

# 3. 运行
python run.py -c config.yml
